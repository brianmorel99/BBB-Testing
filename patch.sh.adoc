    37	git_add () {
    38		git add .
    39		git commit -a -m 'testing patchset'
    40	}
    41	
    42	start_cleanup () {
    43		git="git am --whitespace=fix"
    44	}
    45	
    46	cleanup () {
    47		if [ "${number}" ] ; then
    48			git format-patch -${number} -o ${DIR}/patches/
    49		fi
    50		exit
    51	}
